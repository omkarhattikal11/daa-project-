CASE 1 — Industrial Transport Route Optimization

Algorithm: Dijkstra | SDG 9

About: Finds minimum travel time between industrial zones.

#include <bits/stdc++.h>
using namespace std;

struct Edge {
    int v, w;
};

int main() {
    int n = 6;
    vector<vector<Edge>> graph(n);

    auto addEdge = [&](int u,int v,int w){
        graph[u].push_back({v,w});
        graph[v].push_back({u,w});
    };

    addEdge(0,1,4); addEdge(0,2,2);
    addEdge(1,2,1); addEdge(1,3,5);
    addEdge(2,3,8); addEdge(3,4,6);
    addEdge(4,5,3);

    vector<int> dist(n,1e9);
    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<>> pq;

    dist[0]=0;
    pq.push({0,0});

    while(!pq.empty()){
        auto [d,u]=pq.top(); pq.pop();
        if(d>dist[u]) continue;
        for(auto e:graph[u]){
            if(dist[e.v]>d+e.w){
                dist[e.v]=d+e.w;
                pq.push({dist[e.v],e.v});
            }
        }
    }

    for(int i=0;i<n;i++)
        cout<<"Zone "<<i<<" Distance: "<<dist[i]<<endl;
}

CASE 7 — Fast Facility Search

Algorithm: Trie | SDG 9

#include <bits/stdc++.h>
using namespace std;

struct Trie {
    map<char,Trie*> next;
    bool end=false;
};

void insert(Trie* root,string s){
    for(char c:s){
        if(!root->next[c]) root->next[c]=new Trie();
        root=root->next[c];
    }
    root->end=true;
}

bool search(Trie* root,string s){
    for(char c:s){
        if(!root->next[c]) return false;
        root=root->next[c];
    }
    return root->end;
}

int main(){
    Trie* root=new Trie();
    insert(root,"Hospital");
    insert(root,"Hostel");

    cout<<search(root,"Hospital");
}
